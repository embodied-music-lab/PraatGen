# ============================================================================
# §1  PLUGIN DIRECTORY STRUCTURE
# ============================================================================
#
# Plugins are folders named plugin_[Name] placed in the Praat
# preferences directory (preferencesDirectory$ in scripts). Praat
# runs setup.praat from each plugin folder at startup.
#
# Standard layout:
#
#   plugin_MyPlugin/
#   ├── setup.praat           # Registration (runs at Praat startup)
#   ├── myScript.praat        # Main script(s)
#   ├── procedures/           # Shared procedure files (included)
#   └── scripts/              # Scripts registered via setup.praat
#
# The preferences directory path varies by OS:
#   macOS:   /Users/[name]/Library/Preferences/Praat Prefs/
#   Windows: C:\Users\[name]\Praat\
#   Linux:   /home/[name]/.praat-dir/
#
# In scripts: preferencesDirectory$
# NOTE: preferencesDirectory$ cannot be used inside include directives.
# include is a preprocessor directive — it resolves at parse time,
# before any variables exist. Include paths are always relative to the
# calling script's directory.

# ============================================================================
# §2  SETUP.PRAAT — REGISTRATION COMMANDS
# ============================================================================
#
# setup.praat registers menu items and dynamic action buttons.
# It runs once at Praat startup. It should contain ONLY registration
# commands — no analysis logic, no user dialogs, no object creation.
#
# Two registration commands:
#
#   Add menu command: window$, menu$, title$, after$, depth, script$
#   Add action command: class1$, n1, class2$, n2, class3$, n3,
#       ... title$, after$, depth, script$

# ============================================================================
# §3  Add menu command:
# ============================================================================
#
# Adds a fixed menu item (always visible regardless of selection).
#
# Parameters:
#   window$  — "Objects" or "Picture"
#   menu$    — Menu name: "Open", "Save", "New", "Help", etc.
#   title$   — Menu item text. Use "..." suffix for commands with dialogs.
#              Use "-- label --" for separator lines.
#              Use "" with depth 0 for cascade submenu headers.
#   after$   — Insert after this existing command ("" = append to end)
#   depth    — Submenu nesting level (0 = top level, 1 = submenu child)
#   script$  — Path relative to the plugin directory. "" for submenu
#              headers (no script to run).
#
# EXAMPLES:
#
#   # Top-level cascade header (no script — becomes submenu title):
#   Add menu command: "Objects", "New", "My Plugin", "", 0, ""
#
#   # Separator line within a submenu:
#   Add menu command: "Objects", "New", "-- my section --", "My Plugin", 1, ""
#
#   # Submenu item (depth 1, after the separator):
#   Add menu command: "Objects", "New", "Analyze...",
#       ... "-- my section --", 1, "scripts/analyze.praat"
#
# SUBMENU CASCADE PATTERN:
#   A depth-0 entry with no script creates a cascade header.
#   Depth-1 entries referencing it (via after$) become children.
#   Separator lines ("-- label --") divide sections within the submenu.
#
#   Add menu command: "Objects", "New", "EML Tools", "", 0, ""
#   Add menu command: "Objects", "New", "-- describe --", "EML Tools", 1, ""
#   Add menu command: "Objects", "New", "Describe column...",
#       ... "-- describe --", 1, "scripts/describe.praat"
#   Add menu command: "Objects", "New", "-- compare --",
#       ... "Describe column...", 1, ""
#   Add menu command: "Objects", "New", "Compare groups...",
#       ... "-- compare --", 1, "scripts/compare.praat"

# ============================================================================
# §4  Add action command:
# ============================================================================
#
# Adds a dynamic button that appears only when specific object types
# are selected.
#
# Parameters:
#   class1$  — Required object type (e.g., "Sound", "Table")
#   n1       — Number of class1 objects required (0 = any number,
#              1 = exactly one, 2 = exactly two, etc.)
#   class2$  — Optional second type ("" to skip)
#   n2       — Number of class2 objects required (0 if skipped)
#   class3$  — Optional third type ("" to skip)
#   n3       — Number of class3 objects required (0 if skipped)
#   title$   — Button text. Use "..." suffix for commands with dialogs.
#   after$   — Insert after this existing command ("" = append)
#   depth    — Submenu nesting (0 = top level)
#   script$  — Path relative to plugin directory
#
# EXAMPLES:
#
#   # Appears when exactly 1 Sound is selected:
#   Add action command: "Sound", 1, "", 0, "", 0,
#       ... "Analyze EGG...", "", 0, "scripts/eggAnalysis.praat"
#
#   # Appears when 1 Sound + 1 TextGrid are selected:
#   Add action command: "Sound", 1, "TextGrid", 1, "", 0,
#       ... "Extract measures...", "", 0, "scripts/extractMeasures.praat"
#
#   # Appears when any number of Sounds are selected:
#   Add action command: "Sound", 0, "", 0, "", 0,
#       ... "Batch process...", "", 0, "scripts/batch.praat"
#
#   # Appears when exactly 1 Table is selected:
#   Add action command: "Table", 1, "", 0, "", 0,
#       ... "EML: Describe column...", "", 0,
#       ... "scripts/eml-describe-table.praat"
#
#   # Editor menu entry (TableEditor Edit menu):
#   Add menu command: "TableEditor", "Edit",
#       ... "EML: Edit Table...", "", 0,
#       ... "scripts/eml-edit-table-editor.praat"
#
# Source: PRAAT_DEFINITIVE_CATALOGUE.txt, Praat manual Scripting 8.

# ============================================================================
# §5  INCLUDE PATH BEHAVIOR
# ============================================================================
#
# include is a preprocessor directive. It resolves paths relative to
# the directory of the script that contains the include statement.
#
# For a plugin at plugin_MyPlugin/:
#
#   # In scripts/analyze.praat:
#   include ../procedures/helpers.praat
#   # Resolves to: plugin_MyPlugin/procedures/helpers.praat
#
#   # In setup.praat (at plugin root):
#   # include is NOT typically used in setup.praat — setup only
#   # registers commands, it does not define procedures.
#
# CANNOT use variables in include paths:
#   include 'preferencesDirectory$'/myFile.praat    # WRONG — parse error
#   include ../procedures/myFile.praat              # RIGHT — literal path
#
# For procedure files that need to be shared across plugins, the
# including script must know the relative path at parse time. There
# is no dynamic include mechanism.

# ============================================================================
# §6  PLUGIN-CONFLICT GUARDS
# ============================================================================
#
# When a script might run either as a standalone or as part of a
# plugin (where shared procedures are already loaded via include),
# use variableExists() to avoid double-loading:
#
#   if not variableExists ("emlCoreLoaded")
#       include eml-core-utilities.praat
#   endif
#
# Sentinel pattern: Define a guard variable at the top of each
# procedure file:
#
#   # In eml-core-utilities.praat:
#   emlCoreLoaded = 1
#
# This prevents parse errors from double-include and allows scripts
# to function both standalone and as plugin components.
#
# Source: EML Stats plugin, EML Edit Table plugin.
